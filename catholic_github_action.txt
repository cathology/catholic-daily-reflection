name: Build Daily Reflections

on:
  # Manual trigger
  workflow_dispatch:
  
  # Optional: Run daily at 3 AM UTC to refresh content
  # schedule:
  #   - cron: '0 3 * * *'

permissions:
  contents: write

jobs:
  build:
    runs-on: ubuntu-latest
    
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
        with:
          fetch-depth: 0
      
      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
      
      - name: Install dependencies
        run: |
          cd build-scripts
          npm init -y
          npm install cheerio node-fetch
      
      - name: Run content fetch script
        run: |
          node build-scripts/fetch-data.js
        env:
          NODE_ENV: production
      
      - name: Check for changes
        id: git-check
        run: |
          git diff --exit-code data/entries.json || echo "changed=true" >> $GITHUB_OUTPUT
      
      - name: Commit and push if changed
        if: steps.git-check.outputs.changed == 'true'
        run: |
          git config user.name "GitHub Actions Bot"
          git config user.email "actions@github.com"
          git add data/entries.json
          git commit -m "Update daily reflections content [skip ci]"
          git push
      
      - name: No changes detected
        if: steps.git-check.outputs.changed != 'true'
        run: echo "No changes to entries.json, skipping commit"
